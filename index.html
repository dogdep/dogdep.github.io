Simple configuration

DOGPRO.YML 
defaults:
  deploy_dir: "/var/www/some-project"

plays:

  - name: "NGINX server"
    role: dogpro.nginx

  - name: "NGINX config"
    role: dogpro.nginx_site
    type: symfony

  - name: "Install MySQL"
    role: dogpro.mysql

  - name: "PHP 5.6"
    role: dogpro.php56
    modules:
     - "php-redis"

  - name: "NodeJS"
    role: dogpro.nodejs

  - name: "Redis master"
    role: dogpro.redis

  - name: "Deploy"
    role: dogpro.deploy
    shared_dirs:
      - "app/logs"
      - "uploads"
